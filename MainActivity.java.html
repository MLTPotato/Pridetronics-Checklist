import android.support.v7.app.AppCompatActivity;
import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.ImageView;
import android.widget.TextView;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        //Creates buttons
        final Button finish = findViewById(R.id.FinishBTN);

        //Check Boxes for GamePad
        final CheckBox leftThumb = findViewById(R.id.LeftThumbBox);
        final CheckBox rightThumb = findViewById(R.id.RightThumbBox);
        final CheckBox thumbsLR = findViewById(R.id.ThumbsLRBox);
        final CheckBox leftTrigger = findViewById(R.id.LeftTriggerBox);
        final CheckBox rightTrigger = findViewById(R.id.RightTriggerBox);
        final CheckBox buttonA = findViewById(R.id.ButtonABox);
        final CheckBox buttonB = findViewById(R.id.ButtonBBox);
        final CheckBox buttonX = findViewById(R.id.ButtonXBox);
        final CheckBox buttonY = findViewById(R.id.ButtonYBox);
        final CheckBox leftBumper = findViewById(R.id.LeftBumpBox);
        final CheckBox rightBumper = findViewById(R.id.RightBumpBox);
        final CheckBox backStartButton = findViewById(R.id.BackStartButtonBox);
        final CheckBox thumbsPress = findViewById(R.id.ThumbsPressBox);

        //Check Boxes for Joystick
        final CheckBox joystickUDLR = findViewById(R.id.JoystickUDLRBox);
        final CheckBox throttle = findViewById(R.id.ThrottleBox);
        final CheckBox trigger = findViewById(R.id.TriggerBox);
        final CheckBox button2 = findViewById(R.id.Button2Box);
        final CheckBox button3 = findViewById(R.id.Button3Box);
        final CheckBox button4 = findViewById(R.id.Button4Box);
        final CheckBox button5 = findViewById(R.id.Button5Box);
        final CheckBox button6 = findViewById(R.id.Button6Box);
        final CheckBox button7 = findViewById(R.id.Button7Box);
        final CheckBox button8 = findViewById(R.id.Button8Box);
        final CheckBox button9 = findViewById(R.id.Button9Box);
        final CheckBox button10 = findViewById(R.id.Button10Box);
        final CheckBox button11 = findViewById(R.id.Button11Box);
        final CheckBox button12 = findViewById(R.id.Button12Box);

        //Check Boxes for Auto Behaviors
        final CheckBox auto1 = findViewById(R.id.Auto1Box);
        final CheckBox auto2 = findViewById(R.id.Auto2Box);
        final CheckBox auto3 = findViewById(R.id.Auto3Box);

        //Check Boxes for Systems Check
        final CheckBox cameraShuff = findViewById(R.id.CamboardBox);
        final CheckBox canLeft = findViewById(R.id.CAN_DriveLeftBox);
        final CheckBox canRight = findViewById(R.id.CAN_DriveRightBox);
        final CheckBox canLift = findViewById(R.id.CAN_LiftBox);
        final CheckBox wristUP = findViewById(R.id.WristUPBox);
        final CheckBox wristDown = findViewById(R.id.WristDOWNBox);
        final CheckBox wristUpLim = findViewById(R.id.WristUpperLimBox);
        final CheckBox wristDownLim = findViewById(R.id.WristLowerLimBox);
        final CheckBox liftLowerLim = findViewById(R.id.LiftLowerLimBox);

        //When the finish button is clicked
        finish.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {



                //Creates all of the finish screen background boxes
                final ImageView finishBackgroundBox = findViewById(R.id.finishBackgroundBox);
                final ImageView finishForgroundBox = findViewById(R.id.finishForgroundBox);

                //Creates all of the Gamepad finish screen objects
                final ImageView gamePadBacker = findViewById(R.id.gamePadStatusBacker);
                final TextView gamePadMessage = findViewById(R.id.gamePadMessage);
                final TextView gamePadStatus = findViewById(R.id.gamePadStatus);

                //Creates all of the Joystick finish screen objects
                final ImageView joystickBacker = findViewById(R.id.joystickStatusBacker);
                final TextView joystickMessage = findViewById(R.id.joystickMessage);
                final TextView joystickStatus = findViewById(R.id.joystickStatus);

                //Creates all of the Auto Behavior finish screen objects
                final ImageView autoBehaviorBacker = findViewById(R.id.AutoBehaviorsStatusBacker);
                final TextView autoBehaviorMessage = findViewById(R.id.AutoBehaviorsMessage);
                final TextView autoBehaviorStatus = findViewById(R.id.AutoBehaviorStatus);

                //Creates all of the Systems Check finish screen objects
                final ImageView systemsCheckBacker = findViewById(R.id.SystemsCheckStatusBacker);
                final TextView systemsCheckMessage = findViewById(R.id.SystemsCheckMessage);
                final TextView systemsCheckStatus = findViewById(R.id.SystemsCheckStatus);

                //Creates the finish screen done button
                final Button doneBTN = findViewById(R.id.doneBTN);

                //Sets all the finish screen objects to be visible
                finishBackgroundBox.setVisibility(View.VISIBLE);
                finishForgroundBox.setVisibility(View.VISIBLE);

                joystickBacker.setVisibility(View.VISIBLE);

                autoBehaviorBacker.setVisibility(View.VISIBLE);
                systemsCheckMessage.setVisibility(View.VISIBLE);

                doneBTN.setVisibility(View.VISIBLE);

                // Checks which systems have been completed
                if(rightThumb.isChecked()&& leftThumb.isChecked() && thumbsLR.isChecked() && leftTrigger.isChecked()&& rightTrigger.isChecked()&& buttonA.isChecked()&& buttonB.isChecked()&& buttonX.isChecked() && buttonY.isChecked()&& leftBumper.isChecked()&& rightBumper.isChecked() && backStartButton.isChecked() && thumbsPress.isChecked())
                {
                    gamePadStatus.setText(R.string.nominal);
                    gamePadStatus.setTextColor(getResources().getColor(R.color.colorGreen));
                    gamePadStatus.setVisibility(View.VISIBLE);
                }else
                {
                    gamePadStatus.setText(R.string.error);
                    gamePadStatus.setTextColor(getResources().getColor(R.color.colorErrorRed));
                    gamePadStatus.setVisibility(View.VISIBLE);
                }
                if(joystickUDLR.isChecked() && throttle.isChecked() && trigger.isChecked()&& button2.isChecked()&& button3.isChecked()&& button4.isChecked()&& button5.isChecked()&& button6.isChecked()&& button7.isChecked()&& button8.isChecked()&& button9.isChecked()&& button10.isChecked()&& button11.isChecked()&& button12.isChecked())
                {
                    joystickStatus.setText(R.string.nominal);
                    joystickStatus.setTextColor(getResources().getColor(R.color.colorGreen));
                    joystickStatus.setVisibility(View.VISIBLE);
                }else
                {
                    joystickStatus.setText(R.string.error);
                    joystickStatus.setTextColor(getResources().getColor(R.color.colorErrorRed));
                    joystickStatus.setVisibility(View.VISIBLE);
                }
                if(auto1.isChecked() && auto2.isChecked() && auto3.isChecked())
                {
                    autoBehaviorStatus.setText(R.string.nominal);
                    autoBehaviorStatus.setTextColor(getResources().getColor(R.color.colorGreen));
                    autoBehaviorStatus.setVisibility(View.VISIBLE);
                }else
                {
                    autoBehaviorStatus.setText(R.string.error);
                    autoBehaviorStatus.setTextColor(getResources().getColor(R.color.colorErrorRed));
                    autoBehaviorStatus.setVisibility(View.VISIBLE);
                }
                if(cameraShuff.isChecked() && canLeft.isChecked() && canRight.isChecked() && canLift.isChecked() && wristUP.isChecked() && wristDown.isChecked() && wristUpLim.isChecked() && wristDownLim.isChecked() && liftLowerLim.isChecked())
                {
                    systemsCheckStatus.setText(R.string.nominal);
                    systemsCheckStatus.setTextColor(getResources().getColor(R.color.colorGreen));
                    systemsCheckStatus.setVisibility(View.VISIBLE);
                }else
                {
                    systemsCheckStatus.setText(R.string.error);
                    systemsCheckStatus.setTextColor(getResources().getColor(R.color.colorErrorRed));
                    systemsCheckStatus.setVisibility(View.VISIBLE);
                }

                //When the done button is clicked it hides the finish screen
                doneBTN.setOnClickListener(new View.OnClickListener() {
                    @Override
                    public void onClick(View v) {

                        finishBackgroundBox.setVisibility(View.GONE);
                        finishForgroundBox.setVisibility(View.GONE);
                        gamePadMessage.setVisibility(View.GONE);
                        gamePadBacker.setVisibility(View.GONE);
                        joystickMessage.setVisibility(View.GONE);
                        joystickBacker.setVisibility(View.GONE);
                        doneBTN.setVisibility(View.GONE);
                        joystickStatus.setVisibility(View.GONE);
                        gamePadStatus.setVisibility(View.GONE);
                        autoBehaviorMessage.setVisibility(View.GONE);
                        autoBehaviorBacker.setVisibility(View.GONE);
                        autoBehaviorStatus.setVisibility(View.GONE);
                        systemsCheckMessage.setVisibility(View.GONE);
                        systemsCheckStatus.setVisibility(View.GONE);
                        systemsCheckBacker.setVisibility(View.GONE);
                    }
                });
            }
        });
    }

}
